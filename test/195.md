function GetStar($time) {
    $mon = date('n', $time);
    $day = date('j', $time);
    $star_arr = array(
        array(20, '水瓶座'),
        array(19, '双鱼座'),
        array(21, '白羊座'),
        array(20, '金牛座'),
        array(21, '双子座'),
        array(22, '巨蟹座'),
        array(23, '狮子座'),
        array(23, '处女座'),
        array(23, '天秤座'),
        array(24, '天蝎座'),
        array(23, '射手座'),
        array(22, '摩羯座'), );
    $key = $star_arr[$mon - 1][0] > $day ? ($mon - 2) : ($mon - 1);
    if ($key < 0) {
        $key = 11;
    }
    return $star_arr[$key][1];
}
